  37. function cut_color($color)
    38. 

  38. {
    39. 

  39.     $replace_colors =& $GLOBALS['csstidy']['replace_colors'];
    40. 

  40.     
    41. 

  41.     // rgb(0,0,0) -> #000000 (or #000 in this case later)
    42. 

  42.     if(strtolower(substr($color,0,4)) == 'rgb(')
    43. 

  43.     {
    44. 

  44.         $color_tmp = substr($color,4,strlen($color)-5);
    45. 

  45.         $color_tmp = explode(',',$color_tmp);
    46. 

  46.         for ( $i = 0; $i < count($color_tmp); $i++ )
    47. 

  47.         {
    48. 

  48.             $color_tmp[$i] = trim ($color_tmp[$i]);
    49. 

  49.             if(substr($color_tmp[$i],-1) == '%')
    50. 

  50.             {
    51. 

  51.                 $color_tmp[$i] = round((255*$color_tmp[$i])/100);
    52. 

  52.             }
    53. 

  53.             if($color_tmp[$i]>255) $color_tmp[$i] = 255;
    54. 

  54.         }
    55. 

  55.         $color = '#';
    56. 

  56.         for ($i = 0; $i < 3; $i++ )
    57. 

  57.         {
    58. 

  58.             if($color_tmp[$i]<16) $color .= '0'.dechex($color_tmp[$i]);
    59. 

  59.                 else $color .= dechex($color_tmp[$i]);
    60. 

  60.         }
    61. 

  61.     }
    62. 

  62.     
    63. 

  63.     // Fix bad color names
    64. 

  64.     if(isset($replace_colors[strtolower($color)]))
    65. 

  65.     {
    66. 

  66.         $color = $replace_colors[strtolower($color)];
    67. 

  67.     }
    68. 

  68.     
    69. 

  69.     // #aabbcc -> #abc
    70. 

  70.     if(strlen($color) == 7)
    71. 

  71.     {
    72. 

  72.         $color_temp = strtolower($color);
    73. 

  73.         if($color_temp{0} == '#' && $color_temp{1} == $color_temp{2} && $color_temp{3} == $color_temp{4} && $color_temp{5} == $color_temp{6})
    74. 

  74.         {
    75. 

  75.             $color = '#'.$color{1}.$color{3}.$color{5};
    76. 

  76.         }
    77. 

  77.     }
    78. 

  78.     
    79. 

  79.     switch(strtolower($color))
    80. 

  80.     {
    81. 

  81.         /* color name -> hex code */
    82. 

  82.         case 'black': return '#000';
    83. 

  83.         case 'fuchsia': return '#F0F';
    84. 

  84.         case 'white': return '#FFF';
    85. 

  85.         case 'yellow': return '#FF0';
    86. 

  86.                 
    87. 

  87.         /* hex code -> color name */
    88. 

  88.         case '#800000': return 'maroon';
    89. 

  89.         case '#ffa500': return 'orange';
    90. 

  90.         case '#808000': return 'olive';
    91. 

  91.         case '#800080': return 'purple';
    92. 

  92.         case '#008000': return 'green';
    93. 

  93.         case '#000080': return 'navy';
    94. 

  94.         case '#008080': return 'teal';
    95. 

  95.         case '#c0c0c0': return 'silver';
    96. 

  96.         case '#808080': return 'gray';
    97. 

  97.         case '#f00': return 'red';    
    98. 

  98.     }
    99. 

  99.  
    100. 

  100.     return $color;
    101. 

  101. }

  113.     $number_values =& $GLOBALS['csstidy']['number_values'];

    115. 

  115.  
    116.

  117.     if($property == 'font')
    118. 

  118.     {
    119. 

  119.         $temp = explode('/',$subvalue);
    120. 

  120.     }
    121. 

  121.     else
    122. 

  122.     {
    123. 

  123.         $temp = array($subvalue);
    124. 

  124.     }
    125. 

  125.     for ($l = 0; $l < count($temp); $l++)
    126. 

  126.     {
    127. 

  127.         // continue if no numeric value
    128. 

  128.         if(!(strlen($temp[$l]) > 0 && ( is_numeric($temp[$l]{0}) || $temp[$l]{0} == "+" || $temp[$l]{0} == "-" ) ))
    129. 

  129.         {
    130. 

  130.             continue;
    131. 

  131.         }
    132. 

  132.  
    133. 

  133.         // Fix bad colors
    134. 

  134.         if(in_array($property,$color_values))
    135. 

  135.         {
    136. 

  136.             $temp[$l] = '#'.$temp[$l];
    137. 

  137.         }
    138. 

  138.     
    139. 

  139.         if(floatval($temp[$l]) == 0 && ( is_numeric($temp[$l]{0}) || $temp[$l]{0} == "+" || $temp[$l]{0} == "-" ) )
    140. 

  140.         {
    141. 

  141.             $temp[$l] = 0;
    142. 

  142.         }
    143. 

  143.         elseif(is_numeric($temp[$l]{0}) || $temp[$l]{0} == "+" || $temp[$l]{0} == "-")
    144. 

  144.         {
    145. 

  145.             $unit_found = FALSE;
    146. 

  146.             for( $m = 0, $size_4 = count($units); $m < $size_4; $m++ )
    147. 

  147.             {
    148. 

  148.                 if(strpos(strtolower($temp[$l]),$units[$m]) !== FALSE)
    149. 

  149.                 {
    150. 

  150.                     $temp[$l] = floatval($temp[$l]).$units[$m];
    151. 

  151.                     $unit_found = TRUE;
    152. 

  152.                     break;
    153. 

  153.                 }
    154. 

  154.             }
    155. 

  155.             if(!$unit_found && !in_array($property,$number_values,TRUE))
    156. 

  156.             {
    157. 

  157.                 $temp[$l] = floatval($temp[$l]).'px';
    158. 

  158.             }
    159. 

  159.             else if(!$unit_found)
    160. 

  160.             {
    161. 

  161.                 $temp[$l] = floatval($temp[$l]);
    162. 

  162.             }
    163. 

  163.         }
    164. 

  164.     }
    165. 

  165.     $subvalue = (count($temp) > 1) ? $temp[0].'/'.$temp[1] : $temp[0];
    166. 

  166.     return "$subvalue";
    167. 

  167. }

  235. function explode_ws($sep,$string)
    236. 

  236. {

  237.     $status = 'st';

  238.     $to = '';
    239. 

  239.     
    240. 

  240.     $output = array();
    241. 

  241.     $num = 0;
    242. 

  242.     for($i = 0, $len = strlen($string);$i < $len; $i++)
    243. 

  243.     {
    244. 

  244.         switch($status)
    245. 

  245.         {
    246. 

  246.             case 'st':
    247. 

  247.             if($string{$i} == $sep && !csstidy::escaped($string,$i))
    248. 

  248.             {
    249. 

  249.                 ++$num;
    250. 

  250.             }
    251. 

  251.             elseif($string{$i} == '"' || $string{$i} == '\'' || $string{$i} == '(' && !csstidy::escaped($string,$i))

  252.             {

  253.                 $status = 'str';
    254. 

  254.                 $to = ($string{$i} == '(') ? ')' : $string{$i};
    255. 

  255.                 (isset($output[$num])) ? $output[$num] .= $string{$i} : $output[$num] = $string{$i};
    256. 

  256.             }
    257. 

  257.             else
    258. 

  258.             {
    259. 

  259.                 (isset($output[$num])) ? $output[$num] .= $string{$i} : $output[$num] = $string{$i};
    260. 

  260.             }
    261. 

  261.             break;
    262. 

  262.             
    263. 

  263.             case 'str':

  264.             if($string{$i} == $to && !csstidy::escaped($string,$i))
    265. 

  265.             {
    266. 

  266.                 $status = 'st';
    267. 

  267.             }
    268. 

  268.             (isset($output[$num])) ? $output[$num] .= $string{$i} : $output[$num] = $string{$i};
    269. 

  269.             break;
    270. 

  270.         }
    271. 

  271.     }
    272. 

  272.     
    273. 

  273.     if(isset($output[0]))
    274. 

  274.     {

    276.
